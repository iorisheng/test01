def remove_extra_parentheses(sql):
    # 如果 SQL 中没有括号，则直接返回 SQL
    if '(' not in sql and ')' not in sql:
        return sql
    
    # 找到 SQL 中最里层的一对括号，同时保证这对括号内没有其他括号
    start = sql.rfind('(')
    end = sql.find(')', start)
    if sql[start+1:end].count('(') == sql[start+1:end].count(')') == 0:
        # 如果这对括号中只有一个条件，则可以直接删除这对括号
        if start == 0 and end == len(sql) - 1:
            return remove_extra_parentheses(sql[1:-1])
        elif start == 0:
            return remove_extra_parentheses(sql[start:end+1]) + remove_extra_parentheses(sql[end+1:])
        elif end == len(sql) - 1:
            return remove_extra_parentheses(sql[:start]) + remove_extra_parentheses(sql[start:end+1])
        else:
            return remove_extra_parentheses(sql[:start]) + remove_extra_parentheses(sql[start:end+1]) + remove_extra_parentheses(sql[end+1:])
    else:
        # 如果这对括号中还有其他括号，则继续递归
        return remove_extra_parentheses(sql[:start]) + remove_extra_parentheses(sql[start:end+1]) + remove_extra_parentheses(sql[end+1:])
